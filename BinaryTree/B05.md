#include <iostream>

struct TNODE {
	int key;
	TNODE* pLeft;
	TNODE* pRight;
};
typedef TNODE* TREE;

TNODE* CreateNode(int x) {
    TNODE* p = new TNODE;
    p->key = x;
    p->pLeft = NULL;
    p->pRight = NULL;
    return p;
}

void InsertNode(TREE& T, int x) {
    if (T == NULL) {
        T = CreateNode(x);
    } else {
        if (x < T->key)
            InsertNode(T->pLeft, x);
        else if (x > T->key)
            InsertNode(T->pRight, x);
    }
}
